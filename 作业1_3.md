## 1. bash 脚本
```bash
#!/bin/bash
# 获取目标文件夹路径
echo -n "请输入要读取的文件夹路径："
read DIRECTORY
# 判断用户输入是否为目录
if [ ! -d "$DIRECTORY" ]; then
    echo "错误：目录 '$DIRECTORY' 不存在！"
    exit 1
fi
# 创建用于存储结果的文本文件
> filenames.txt
> dirname.txt
# 遍历目标目录下的所有内容
for i in "$DIRECTORY"/*;
do
    # 防止空目录导致通配符(*)作为字面量被处理
    [ -e "$i" ] || continue
    # 判断是否为文件
    if [ -f "$i" ]; then
        basename "$i" >> filenames.txt
    # 判断是否为目录
    elif [ -d "$i" ]; then
        basename "$i" >> dirname.txt
    fi
done
# 提示用户已存储
echo "文件名已保存到 filenames.txt，目录名已保存到 dirname.txt。"
exit 0
```
